/******************************************
*** 功能：按下按键，LED灯左移一位。
******************************************/
#include<pic.h>              	
void delay();              		
void delay_mm() ;  
void Keyscan();

char event = 0;
char count = 0;
 
//主函数
void main()                 
{
   TRISC = 0;  		//设置portc为输出
   TRISB = 0x01;  	//设置portb0为输入
   PORTC = 0x01; 

   while (1)         	
   {
       Keyscan();
       if(event)
       {
           event = 0;
           PORTC = PORTC << 1;  
           count++;
           if(count == 8)
           {
               count = 0;
               PORTC = 0x01;
           }    
       }
   }
}

//小延时函数
void delay_mm()              
{
   int t;              
   for (t = 0xfff;t--;);
}

//按键检测函数
void Keyscan()
{   
   // 如果按键第一次被按下
   if(RB0==1)
   {
      //延时消抖
      delay_mm(); 
      if(RB0==1)
	 {
	    // 确认按键被按下后，事件标志位置1
	    event = 1;
	    // 等待按键被松开
	    while(RB0);
	    return;
	 }
   }
}
